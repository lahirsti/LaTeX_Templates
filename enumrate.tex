\chapter{Lists}
When you want to itemize or enumerate.

\section{Itemize}
\begin{itemize}
    \item Item 1
    \item Item 2
    \begin{itemize}
        \item Sub item 1
        \item Sub item 2
    \end{itemize}
    \item Item 3
\end{itemize}

\section{Enumerate}
\begin{enumerate}
    \item Item 1
    \item Item 2
    \begin{enumerate}
        \item Sub item 1
        \item Sub item 2 
    \end{enumerate}
    \item Item 3
\end{enumerate}

\section{Combo}
\begin{enumerate}
    \item Item 1
    \item Item 2
    \begin{itemize}
        \item Sub item 1
        \item Sub item 2 
    \end{itemize}
    \item Item 3
\end{enumerate}

or 

\begin{itemize}
    \item Item 1
    \item Item 2
    \begin{enumerate}
        \item Sub item 1
        \item Sub item 2 
    \end{enumerate}
    \item Item 3
\end{itemize}
